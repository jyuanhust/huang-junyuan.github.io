# 剑指 Offer 05. 替换空格

class Solution {
public:
    string replaceSpace(string s) {
        for (int i = 0; i < s.length(); i++) {
            if (s[i] == ' ') {
                //s[i] = '$';
                string temp = "%20";
                s.replace(i, 1, temp);
                i += temp.length() - 1;
            }
        }
        return s;
    }
};

class Solution {
public:
    string replaceSpace(string s) {
        int count = 0, len = s.size();
        // 统计空格数量
        for (char c : s) {
            if (c == ' ') count++;
        }
        // 修改 s 长度
        s.resize(len + 2 * count);
        // 倒序遍历修改
        for(int i = len - 1, j = s.size() - 1; i < j; i--, j--) {
            if (s[i] != ' ')
                s[j] = s[i];
            else {
                s[j - 2] = '%';
                s[j - 1] = '2';
                s[j] = '0';
                j -= 2;
            }
        }
        return s;
    }
};

# 剑指 Offer 06. 从尾到头打印链表

使用 algorithm 头文件中的 reverse 函数，

class Solution {
public:
    vector<int> reversePrint(ListNode* head) {
        stack<int> stk;
        while(head != nullptr) {
            stk.push(head->val);
            head = head->next;
        }
        vector<int> res;
        while(!stk.empty()) {
            res.push_back(stk.top());
            stk.pop();
        }
        return res;
    }
};
